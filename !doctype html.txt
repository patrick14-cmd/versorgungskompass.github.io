<script>

function el(id){
  return document.getElementById(id);
}

function value(id){
  return el(id).value.trim();
}

function selectedGoals(){

  return Array.from(
    document.querySelectorAll('input[name="ziel"]:checked')
  )
  .map(x => x.parentNode.textContent.trim())
  .join(', ') || 'Keine Auswahl';
}

function buildSummary(){

  return `
Versorgungskompass Anfrage

Name: ${value('name')}
E-Mail: ${value('email')}
Telefon: ${value('phone')}
PLZ / Ort: ${value('city')}

Versorgungsziele:
${selectedGoals()}
`;
}

function clearMessages(){

  el('error').style.display='none';
  el('ok').style.display='none';
}

function showError(msg){

  const box = el('error');

  box.textContent = msg;
  box.style.display = 'block';
}

function validateRequired(){

  clearMessages();

  if(!value('name') || !value('email')){

    showError('Bitte mindestens Name und E-Mail ausfüllen.');
    return false;
  }

  return true;
}

function showSummary(){

  clearMessages();

  const p = el('preview');

  p.textContent = buildSummary();
  p.style.display = 'block';
}

async function copySummary(){

  clearMessages();

  const text = buildSummary();

  showSummary();

  try{

    await navigator.clipboard.writeText(text);

    el('ok').textContent =
      'Zusammenfassung wurde kopiert.';

    el('ok').style.display = 'block';

  }catch(e){

    showError(
      'Kopieren wurde vom Browser blockiert.'
    );
  }
}

function sendMail(){

  if(!validateRequired()) return;

  const text = buildSummary();

  showSummary();

  el('ok').textContent =
    'Das E-Mail-Programm wird geöffnet.';

  el('ok').style.display='block';

  const subject =
    encodeURIComponent('Versorgungskompass Anfrage');

  const body =
    encodeURIComponent(text);

  window.location.href =
    'mailto:?subject=' + subject + '&body=' + body;
}
